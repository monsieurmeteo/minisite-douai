-- 1. Création de la table principale
CREATE TABLE IF NOT EXISTS observations_horaire (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    station_id TEXT NOT NULL,
    timestamp TIMESTAMPTZ NOT NULL,
    t REAL,        -- Température
    td REAL,       -- Point de rosée
    u INT,         -- Humidité
    ff REAL,       -- Vent moyen
    fxi REAL,      -- Rafales
    dd INT,        -- Direction du vent
    rr1 REAL,      -- Précipitations (1h)
    pres REAL,     -- Pression
    created_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(station_id, timestamp)
);

-- 2. Index de performance
CREATE INDEX IF NOT EXISTS idx_obs_hr_station_time ON observations_horaire (station_id, timestamp DESC);

-- 3. Sécurité (Tout le monde lit, seul le robot écrit)
ALTER TABLE observations_horaire ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Lecture publique" ON observations_horaire FOR SELECT USING (true);
CREATE POLICY "Ecriture système" ON observations_horaire FOR INSERT WITH CHECK (auth.role() = 'service_role');

-- 4. Table pour stocker votre Token Météo France
CREATE TABLE IF NOT EXISTS api_secrets (
    provider text PRIMARY KEY,
    access_token text NOT NULL,
    updated_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE api_secrets ENABLE ROW LEVEL SECURITY;
